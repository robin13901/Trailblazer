```dart
/// Reverse lookup by OSM relation id (across all levels — ids are globally
/// unique, RESEARCH.md line 491). Returns null if not loaded or not found.
/// Cheap linear scan over ~20K regions (same memory posture as regionAt).
AdminRegion? regionByOsmId(int osmId) {
  final byLevel = _byLevel;
  if (byLevel == null) return null;
  for (final bucket in byLevel.values) {
    for (final r in bucket) {
      if (r.osmId == osmId) return r;
    }
  }
  return null;
}
```

(b) Create `region_browser_provider.dart`:

```dart
// regionSearchQueryProvider: the search text (StateProvider<String>, default '').
final regionSearchQueryProvider = StateProvider<String>((_) => '');

// regionBrowserProvider: FutureProvider<List<RegionCoverage>> — the FULL
// coverage-gated list, sorted % desc. (Filtering by search happens in the
// derived provider below so typing doesn't re-hit the DB.)
final regionBrowserProvider = FutureProvider<List<RegionCoverage>>((ref) async {
  final lookup = ref.watch(adminRegionLookupProvider);
  await lookup.ensureLoaded();                       // MAIN isolate (Pitfall 1)
  final rows = await ref.watch(coverageCacheDaoProvider).getAllWithCoverage();
  final out = <RegionCoverage>[];
  for (final row in rows) {
    final osmId = int.tryParse(row.regionId);
    if (osmId == null) continue;
    final region = lookup.regionByOsmId(osmId);
    if (region == null) continue;                    // stale row w/o polygon
    if (region.adminLevel == 2) continue;            // exclude Deutschland from browser (RESEARCH line 273)
    out.add(RegionCoverage(
      osmId: osmId,
      adminLevel: region.adminLevel,
      name: region.nameDe ?? region.name,
      drivenLengthM: row.drivenLengthM,
      totalLengthM: row.totalLengthM,
    ));
  }
  out.sort((a, b) => b.percent.compareTo(a.percent)); // %-desc (CONTEXT line 35)
  return out;
});

// regionBrowserFilteredProvider: Provider<List<RegionCoverage>> — applies the
// fuzzy search on top of the loaded list. Pure-Dart rank (RESEARCH lines 89-92):
//   starts-with match ranks above contains-anywhere; empty query = full list.
final regionBrowserFilteredProvider = Provider<List<RegionCoverage>>((ref) {
  final all = ref.watch(regionBrowserProvider).value ?? const [];
  final q = ref.watch(regionSearchQueryProvider).trim().toLowerCase();
  if (q.isEmpty) return all;
  final starts = <RegionCoverage>[];
  final contains = <RegionCoverage>[];
  for (final r in all) {
    final n = r.name.toLowerCase();
    if (n.startsWith(q)) {
      starts.add(r);
    } else if (n.contains(q)) {
      contains.add(r);
    }
  }
  return [...starts, ...contains];   // ranked; each sub-list already %-desc
});
```

Plain `Provider`/`FutureProvider`/`StateProvider` — NO codegen. Package imports.
