```dart
static const String kRawGpsRetentionDays = 'raw_gps_retention_days';

/// Retention window in days. null = keep forever (no sweep). 0 = delete raw
/// points immediately after matching. Defaults to 30 when unset.
Future<int?> getRawGpsRetentionDays() async {
  // Distinguish "unset" (return default 30) from "forever" (explicit null token).
  // Store the forever choice as a sentinel string via a companion bool key, OR
  // adopt: key absent => default 30; key present with value -1 => forever.
  final v = await _prefs.getInt(kRawGpsRetentionDays);
  return v == null ? 30 : (v < 0 ? null : v);
}

/// days == null => forever (store -1 sentinel so it is distinguishable from
/// "never set" which defaults to 30). 0/30/365 stored verbatim.
Future<void> setRawGpsRetentionDays(int? days) =>
    _prefs.setInt(kRawGpsRetentionDays, days ?? -1);
```
(Choose the sentinel approach above so "unset -> 30 default" and "explicit forever -> null" are BOTH representable — the research's remove-key approach loses the default/forever distinction. Document this choice.)
